    import Color from 'color';
import React from 'react';
import { connect } from 'react-redux';
import { resolve } from 'styled-jsx/css';

import Button from '../controls/Button';
import Logo from './svg/Logo';

import config from '../../../config';


/**
 * A navigation layout component that links to the main areas of the application.
 * @param {Object} props - React Component props.
 * @param {string} props.headerHeight - Header height mapped from Redux store.
 * @return {React.Component}
 */
export function Navigation({ headerHeight }) {
  const baseColor = new Color(config.brandColor);
  const { className, styles } = resolve`
    a,
    button {

    }
  `;

  const links = [
    { name: 'Pokémon', to: '/', icon: Logo, color: baseColor },
    { name: 'Attacks', to: '/attacks', icon: Logo, color: baseColor.hue(30) },
    { name: 'Items', to: '/items', icon: Logo, color: baseColor.hue(60) },
    { name: 'Abilities', to: '/abilities', icon: Logo, color: baseColor.hue(120) },
    { name: 'Berries', to: '/berries', icon: Logo, color: baseColor.hue(180) },
    { name: 'Tools', to: '/tools', icon: Logo, color: baseColor.hue(210) },
  ];

  return (
    <nav>
      <ul>
        {
          links.map((link, i) => (
            <li key={ i }>
              <Button
                className={ className }
                kind="flat"
                link="nav"
                safeAreaInset="left"
                size={ headerHeight }
                iconColor={ link.color }
                icon={ link.icon }
                to={ link.to }
                vertical
                max
              >
                {/* <link.icon fill={ link.color } /> */}
                {/* <span>{ link.name }</span> */}
                {/* { link.name } */}
              </Button>
            </li>
          ))
        }
      </ul>

      { styles }

      <style jsx>{ `
        nav {
          position: fixed;
          z-index: 1;
          display: flex;
          flex-direction: column;
          box-sizing: border-box;
          width: ${headerHeight}px;
          width: calc(${headerHeight}px + var(--safe-area-inset-left));
          height: 100%;
          padding-top: ${headerHeight}px;
          padding-bottom: var(--safe-area-inset-bottom);
          background-color: ${config.theme.dark.appBar};
          color: ${config.theme.dark.textPrimary};
          filter: drop-shadow(0 3px 6px rgba(0,0,0,0.16)) drop-shadow(0 3px 6px rgba(0,0,0,0.23));

          ul {
            flex: 1;
            overflow-y: auto;

            li {
              /* padding-left: env(safe-area-inset-left); */
            }
          }
        }
      ` }</style>
    </nav>
  );
}

export default connect(state => ({
  headerHeight: state.ui.metrics.headerHeight,
}), null)(Navigation);
