    import Color from 'color';
import PropTypes from 'prop-types';
import React, { Component } from 'react';
import anime from 'animejs';
import { resolve } from 'styled-jsx/css';

const pointOffset = 6.335;
const outerHandleOffset = 70.835;
const innerHandleOffset = 583.566;
const getCircleBezier = n => n * 0.552284749831;
const getOuterCurveBezier = n => n * 0.60546875;
const getInnerCurveBezier = n => n * 0.63931719;

const paths = {

export default class Logo extends Component {
  constructor() {
    super();

    this.refPathTop = React.createRef();
    this.refPathMiddle = React.createRef();
    this.refPathBottom = React.createRef();
    this.animateLogo = this.animateLogo.bind(this);
  }

  animateLogo(ref, path) {
    anime.remove(ref.current);
    anime({
      targets: ref.current,
      d: path,
      easing: 'easeInOutQuad',
      duration: 200,
    });
  }

  render() {
    const { className, styles } = resolve`
      svg {
        height: 100%;
        fill: ${this.props.fill};
      }
    `;

    let customClass = className;
    if (this.props.className) customClass += ` ${this.props.className}`;

    return (
      <svg viewBox="0 0 2560 2560" className={ customClass }>
        <path ref={ this.refPathTop } className="top" d={ paths.top.default } />
        <path ref={ this.refPathMiddle } className="middle" d={ paths.middle.default } />
        <path ref={ this.refPathBottom } className="bottom" d={ paths.bottom.default } />

        { styles }
      </svg>
    );
  }

  componentDidUpdate(prevProps) {
    if (!prevProps.hover && this.props.hover) {
      this.animateLogo(this.refPathTop, paths.top.menu);
      this.animateLogo(this.refPathMiddle, paths.middle.menu);
      this.animateLogo(this.refPathBottom, paths.bottom.menu);
    } else if (prevProps.hover && !this.props.hover) {
      this.animateLogo(this.refPathTop, paths.top.default);
      this.animateLogo(this.refPathMiddle, paths.middle.default);
      this.animateLogo(this.refPathBottom, paths.bottom.default);
    }
  }
}

Logo.propTypes = {
  className: PropTypes.string,
  fill: PropTypes.oneOfType([PropTypes.string, PropTypes.instanceOf(Color)]),
  hover: PropTypes.bool,
};
